const marquee = document.getElementById('marquee');
let speed = 0.3; // –®–≤–∏–¥–∫—ñ—Å—Ç—å (—á–∏–º –±—ñ–ª—å—à–µ —á–∏—Å–ª–æ, —Ç–∏–º —à–≤–∏–¥—à–µ)
let offset = 0;

// 1. –ö–ª–æ–Ω—É—î–º–æ –∫–æ–Ω—Ç–µ–Ω—Ç –¥–ª—è —Å—Ç–≤–æ—Ä–µ–Ω–Ω—è —ñ–ª—é–∑—ñ—ó –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ
marquee.innerHTML += marquee.innerHTML; 

function animate() {
    offset -= speed;
    
    // 2. –ö–æ–ª–∏ –ø–µ—Ä—à–∞ –ø–æ–ª–æ–≤–∏–Ω–∞ (–æ—Ä–∏–≥—ñ–Ω–∞–ª—å–Ω—ñ 3 —Ñ–æ—Ç–æ) –ø–æ–≤–Ω—ñ—Å—Ç—é –ø—ñ—à–ª–∞ –≤–ª—ñ–≤–æ
    // –º–∏ —Å–∫–∏–¥–∞—î–º–æ offset –Ω–∞ 0, —â–æ–± –ø–æ—á–∞—Ç–∏ —Ü–∏–∫–ª –∑–∞–Ω–æ–≤–æ –Ω–µ–ø–æ–º—ñ—Ç–Ω–æ
    if (Math.abs(offset) >= marquee.scrollWidth / 2) {
        offset = 0;
    }
    
    marquee.style.transform = `translateX(${offset}px)`;
    
    requestAnimationFrame(animate);
}

// –ó–∞–ø—É—Å–∫–∞—î–º–æ –∫–æ–Ω–≤–µ—î—Ä
animate();

// –û–ø—Ü—ñ–æ–Ω–∞–ª—å–Ω–æ: –∑—É–ø–∏–Ω–∫–∞ –ø—Ä–∏ –Ω–∞–≤–µ–¥–µ–Ω–Ω—ñ –º–∏—à–∫–∏
marquee.addEventListener('mouseenter', () => speed = 0);
marquee.addEventListener('mouseleave', () => speed = 1);
